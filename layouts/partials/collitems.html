{{- /* Returns the photos in a collection as a slice of lightbox-ready dicts. Call via partialCached so each collection's photo scan runs once per build: {{ $items := partialCached "collitems.html" (dict "slug" $slug "base" $base) $slug }} Each item: { thumb, avif, webp, ar, title, meta }. (ar = width/height, for justified layout) */ -}} {{- $slug := .slug -}} {{- $base := .base -}} {{- $items := slice -}} {{- range $k, $p := hugo.Data.photos -}} {{- if and $p.thumb (in (or $p.collections slice) $slug) -}} {{- $ar := 1.5 -}} {{- if and $p.width $p.height (gt $p.height 0) -}}{{- $ar = div (float $p.width) $p.height -}}{{- end -}} {{- $hood := $p.neighborhood | default "" -}} {{- $items = $items | append (dict "thumb" (printf "%s%s" $base $p.thumb) "avif" (printf "%s%s" $base $p.display_avif) "webp" (printf "%s%s" $base $p.display_webp) "ar" $ar "title" (cond (gt (len $hood) 0) (printf "%s, %s" $hood $p.city) $p.city) "meta" (or $p.tag_notes "")) -}} {{- end -}} {{- end -}} {{- return $items -}}
