{{- /* Returns the photos in a collection as a slice of lightbox-ready dicts. Membership is denormalized at deploy time into data/index.json by scripts/photos/build_index.py, so this is an O(1) map lookup + O(k) base-prefix pass instead of scanning the whole manifest once per collection. Still call via partialCached so each slug builds once per render: {{ $items := partialCached "collitems.html" (dict "slug" $slug "base" $base) $slug }} Each item: { thumb, avif, webp, ar, title, meta }. (ar = width/height, for justified layout) */ -}} {{- $slug := .slug -}} {{- $base := .base -}} {{- $items := slice -}} {{- range (index (index hugo.Data.index "collections") $slug) -}} {{- $items = $items | append (dict "thumb" (printf "%s%s" $base .thumb) "avif" (printf "%s%s" $base .avif) "webp" (printf "%s%s" $base .webp) "ar" .ar "title" .title "meta" .meta) -}} {{- end -}} {{- return $items -}}
